#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
ll x;
void solve(ll n)
{
    if(n == 0) return;
    ll bit = n & 1;
    solve(n>>1);
    cout<<bit;
}

int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {

        ll n;
        scanf("%lld",&n);
        solve(n);
        puts("");
    }
}
